Camel DSL

			
@Override
public void configure() throws Exception {
	rest("/aanvragen").description("Signalisatie grote werven aanvraag service")
	  .consumes("application/json").produces("application/json")
	  .get("/{id}").description("Zoek aanvraag via id").outType(Request.class)
		.param().name("id").type(path).description("De id van de op te halen aanvraag").dataType("int").endParam()
		.to("bean:requestService?method=getRequest(${header.id})")
	  .put().description("Wijzig een aanvraag").type(Request.class)
		.param().name("body").type(body).description("De gewijzigde aanvraag").endParam()
		.to("bean:requestService?method=updateRequest");
}

Camel DSL + Activiti component

				
@Override
public void configure() throws Exception {
	rest("/aanvragen”)…
	  .post().description(“Maak een nieuwe aanvraag aan”).type(Request.class)
			.transacted().bean(RequestService.class, “createRequest(${body})”)
				.to(activity:aanvraagSignalisatieGroteWerven);
}
